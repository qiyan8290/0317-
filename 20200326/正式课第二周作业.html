function fn(){

   let result = 0;

   for(var i = 0; i < arguments.length; i++){

       let item = arguments[i];

       item = Number(item);

       if(!isNaN(item)){

           result += item;

       }

   }

   return function (num){

       num = Number(num);

       return !isNaN(num) ? result + num : null;

   };

};

let res = fn(1,2)(3);

console.log(res); //=>6  1+2+3

Number.prototype.plus = function plus(num1){

   num1 = Number(num1);

   num1 = isNaN(num1) ? 0 : num1;

   return this + num1;

};

Number.prototype.minus = function minus(num2){

   num2 = Number(num2);

   num2 = isNaN(num2) ? 0 : num2;

   return this - num2;

};

Array.prototype.unique = function unique(){

    let obj = {};

    for(var i = 0; i < this.length; i++){

        let item = this[i];

        if(typeof obj[item] !== "undefined"){

            this[i] = this[this.length - 1];

            this.length--;

            i--;

            continue;

        }

        obj[item] = item;

    }

    obj = null;

    return this;

};

let ary = [12,23,12,13,13,12,23,14,8];

ary.unique().sort((a,b)=>a-b);
